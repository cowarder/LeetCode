class Solution {
public:
    bool isPalindrome(int x) {
        if(x<0)return false;
        if(x/10==0)return true;
        int c=x;
        int a=x;
        int j=1;
        while(a/10)
        {
            j*=10;
            a/=10;
        }
        int b=0;
        while(x/10!=0)
        {
            b+=(x%10)*j;
            x/=10;
            j/=10;
        }
        b+=x;
        if(c==b)return true;
        else return false;
    }
};
