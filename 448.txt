class Solution {
public:
	vector<int> findDisappearedNumbers(vector<int>& nums) {
		vector<int> a(nums.size());
		for (int j = 0; j < nums.size(); j++)
			a[j] = 0;
		for (int i = 0; i < nums.size(); i++) {
			a[nums[i] - 1]++;
		}
		for (int p = 0; p < nums.size(); p++) {
			if (a[p] == 0)
				a.push_back(p + 1);
		}
		a.begin() += nums.size();
		a.erase(a.begin(), a.begin() + nums.size());
		return a;
	}
};
